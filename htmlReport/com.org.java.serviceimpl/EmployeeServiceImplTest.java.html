        sampleEntity = new Employee(1L, "John Doe", 30, 50000.0, "john@example.com",
                "New York", "Java", "Project A", 123456789012L, "ABCDE1234F", 9876543210L);

        sampleDtoList = Arrays.asList(sampleDto);
        sampleEntityList = Arrays.asList(sampleEntity);
    }


    @Test
    void testAddEmployeeDetails_EmptyList() {
        // Arrange
        List<EmployeeDto> emptyList = Collections.emptyList();
        try (MockedStatic<EmployeeMapper> mockedMapper = Mockito.mockStatic(EmployeeMapper.class)) {
            when(employeeRepository.saveAll(anyList())).thenReturn(Collections.emptyList());

            // Act
            List<EmployeeDto> result = employeeService.addEmployeeDetails(emptyList);

            // Assert
            assertNotNull(result);
            assertTrue(result.isEmpty());
            verify(employeeRepository, times(1)).saveAll(anyList());
        }
    }


    @Test
    void testFetchAllEmployeeDetails_Empty() {
        // Arrange
        try (MockedStatic<EmployeeMapper> mockedMapper = Mockito.mockStatic(EmployeeMapper.class)) {
            when(employeeRepository.findAll()).thenReturn(Collections.emptyList());

            // Act
            List<EmployeeDto> result = employeeService.fetchAllEmployeeDetails();

            // Assert
            assertNotNull(result);
            assertTrue(result.isEmpty());
            verify(employeeRepository, times(1)).findAll();
        }
    }

    @Test
    void testFindByEmployeeId_NotFound() {
        // Arrange
        Long empId = 999L;
        when(employeeRepository.findById(empId)).thenReturn(Optional.empty());

        // Act & Assert
        NoIdFoundException exception = assertThrows(NoIdFoundException.class, () -> {
            employeeService.findByEmployeeId(empId);
        });
        assertEquals("Employee not found with id: ", exception.getErrorMessage());
        assertEquals("404", exception.getErrorCode());
        verify(employeeRepository, times(1)).findById(empId);
    }
